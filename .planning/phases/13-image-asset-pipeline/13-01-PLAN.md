    Create `data/imageMap.ts` following the project's established `Record<EnumType, T>` pattern (see archetypes.ts, deathEndings.ts for reference).

    The file must:

    1. **Export a `slugifyIncident` function** that converts incident names to URL-safe slugs:
       ```typescript
       export function slugifyIncident(incident: string): string {
         return incident
           .toLowerCase()
           .replace(/[^a-z0-9]+/g, "-")
           .replace(/^-|-$/g, "");
       }
       ```

    2. **Export type aliases** (defined in imageMap.ts, NOT in types.ts — keeps image concerns separate):
       ```typescript
       export type IncidentSlug = string;

       export type OutcomeConsequenceType =
         | "BUDGET_CRATER"
         | "PR_DISASTER"
         | "LEGAL_HAMMER"
         | "TEAM_REVOLT"
         | "AUDIT_DOOM"
         | "CAREER_OVER"
         | "REGULATORY_NUKE"
         | "DATA_BREACH";
       ```

    3. **Build INCIDENT_IMAGES dynamically** by importing all card decks, extracting unique `realWorldReference.incident` values, slugifying them, and mapping to paths:
       ```typescript
       import { ALL_ROLE_CARDS } from "./cards";
       // Or import each deck individually if no aggregate export exists

       // Extract unique incidents from all cards
       const uniqueIncidents = new Set<string>();
       for (const card of allCards) {
         if (card.realWorldReference?.incident) {
           uniqueIncidents.add(card.realWorldReference.incident);
         }
       }

       export const INCIDENT_IMAGES: Record<string, string> = Object.fromEntries(
         [...uniqueIncidents].map((incident) => {
           const slug = slugifyIncident(incident);
           return [slug, `/images/incidents/${slug}.webp`];
         })
       );
       ```
       This produces ~118 entries keyed by slug (e.g., "samsung-chatgpt-code-leak" → "/images/incidents/samsung-chatgpt-code-leak.webp"). Cards sharing the same incident automatically share one image.
