import TextOnGif from "text-on-gif";
import * as fs from "node:fs";
import * as path from "node:path";
import templates from "../data/templates/gif-templates.json";

export interface OverlayOptions {
  templateId: string;
  text: string;
  fontSize?: string;
  fontColor?: string;
  position?: "top" | "middle" | "bottom";
  getAsBuffer?: boolean;
}

export interface TemplateInfo {
  name: string;
  textZones: number;
  bestFor: string[];
  url?: string;
}

/**
 * Find a template by ID across all categories
 */
export function getTemplateById(id: string): TemplateInfo | undefined {
  for (const category of Object.values(templates.templates)) {
    const found = category.find((t) => t.id === id);
    if (found) {
      return {
        name: found.name,
        textZones: found.text_zones,
        bestFor: found.best_for,
      };
    }
  }
  return undefined;
}

/**
 * Get templates by incident pattern
 */
export function getTemplatesByPattern(
  pattern: string
): Array<{ id: string; name: string }> {
  const mapping = templates.incident_mapping as Record<string, string[]>;
  const templateIds = mapping[pattern] || [];
  return templateIds.map((id) => {
    const template = getTemplateById(id);
    return {
      id,
      name: template?.name || id,
    };
  });
}

/**
 * Get all available template IDs
 */
export function getAllTemplateIds(): string[] {
  const ids: string[] = [];
  for (const category of Object.values(templates.templates)) {
    for (const t of category) {
      ids.push(t.id);
    }
  }
  return ids;
}

/**
 * Add text to a GIF/template
 * @param options - Overlay configuration
 * @returns Buffer or writes to file
 */
export async function addTextToGif(
  options: OverlayOptions
): Promise<Buffer | null> {
  const {
    templateId,
    text,
    fontSize = "24px",
    fontColor = "white",
    position = "bottom",
    getAsBuffer = true,
  } = options;

  // For now, use a placeholder GIF URL since we don't have template URLs
  // In production, this would fetch from Imgflip or use cached templates
  const gifUrl = `https://i.imgflip.com/1bij.jpg`; // Placeholder - would be dynamic

  try {
    const gif = new TextOnGif({
      file_path: gifUrl,
      font_size: fontSize,
      font_color: fontColor,
      stroke_color: "black",
      stroke_width: 2,
    });

    const result = await gif.textOnGif({
      text,
      alignment_x: "middle",
      alignment_y: position,
      get_as_buffer: getAsBuffer,
    });

    return result;
  } catch (error) {
    console.error("Error adding text to GIF:", error);
    return null;
  }
}

/**
 * Batch process multiple templates with different text
 */
export async function batchOverlay(
  tasks: Array<{ templateId: string; text: string; outputPath?: string }>
): Promise<Array<{ templateId: string; success: boolean; buffer?: Buffer }>> {
  const results = [];

  for (const task of tasks) {
    const buffer = await addTextToGif({
      templateId: task.templateId,
      text: task.text,
    });

    results.push({
      templateId: task.templateId,
      success: buffer !== null,
      buffer: buffer || undefined,
    });

    // Write to file if path provided
    if (task.outputPath && buffer) {
      const dir = path.dirname(task.outputPath);
      fs.mkdirSync(dir, { recursive: true });
      fs.writeFileSync(task.outputPath, buffer);
    }
  }

  return results;
}
